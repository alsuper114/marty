Marty helps you build web applications that follow the Flux architecture. It makes it easy to create stores, bind them to views and get data from your server.

Underscore provides over 100 functions that support both your favorite workaday functional helpers: map, filter, invoke — as well as more specialized goodies: function binding, javascript templating, creating quick indexes, deep equality testing, and so on.

Installation

  • Node.js npm install marty
  • Bower bower install marty

State Mixin


      
var UserState = Marty.createStateMixin({
  required: ['users'],
  listenTo: [UserStore],
  getState: function () {
    return {
      users: UserStore.getAll()
    }
  }
});

Marty.createStateMixin(UserStore);

Marty.createStateMixin({
  users: UserStore,
  whatever: OtherStore
});

var UserState = {
  listenTo: [UserStore],
  getState: function () {
    return {
      user: UserStore.getUser(this.props.id)
    };
  }
};

var OtherState = {
  listenTo: [OtherStore],
  getState: function () {
    return {
      store: OtherStore.getAll()
    }
  }
};

Marty.createStateMixin(UserState, OtherState)

var Users = React.createClass({
